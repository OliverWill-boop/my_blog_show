#### 密码门

- 底层部分

```
unsigned char Key_Read()
{
    unsigned char temp = 0;  // 初始值必须为0，表示无按键按下
    
    // 第一行扫描 (验证密码、数字1、4、7)
    P44 = 0;    P42 = 1;    P35 = 1;    P34 = 1;
    if (P33 == 0)    temp = 4;     // 验证密码按键
    if (P32 == 0)    temp = 101;   // 数字1按键
    if (P31 == 0)    temp = 104;   // 数字4按键
    if (P30 == 0)    temp = 107;   // 数字7按键
    
    // 第二行扫描 (数字0、2、5、8)
    P44 = 1;    P42 = 0;    P35 = 1;    P34 = 1;
    if (P33 == 0)    temp = 100;   // 数字0按键
    if (P32 == 0)    temp = 102;   // 数字2按键
    if (P31 == 0)    temp = 105;   // 数字5按键
    if (P30 == 0)    temp = 108;   // 数字8按键
    
    // 第三行扫描 (清空、数字3、6、9)
    P44 = 1;    P42 = 1;    P35 = 0;    P34 = 1;
    if (P33 == 0)    temp = 12;    // 清空按键
    if (P32 == 0)    temp = 103;   // 数字3按键
    if (P31 == 0)    temp = 106;   // 数字6按键
    if (P30 == 0)    temp = 109;   // 数字9按键
    
    // 第四行扫描 (预留按键)
    P44 = 1;    P42 = 1;    P35 = 1;    P34 = 0;
    if (P33 == 0)    temp = 16;    // 预留按键
    if (P32 == 0)    temp = 17;    // 预留按键
    if (P31 == 0)    temp = 18;    // 预留按键
    if (P30 == 0)    temp = 19;    // 预留按键
    
    return temp;  // 返回按键值
}
```

```
unsigned char current_passwd[8] = {1, 2, 3, 4, 5, 6, 7, 8};  // 系统设定的正确密码
unsigned char input_passwd[8] = {10,10,10,10,10,10,10,10};   // 用户输入的密码缓存
unsigned char input_passwd_index = 0;                        // 当前输入密码的位置索引
bit passwd_error = 0;                                        // 密码错误标志位


// 密码键盘处理函数
void Key_Proc()
{
    static unsigned char Key_Val, Key_Down, Key_Up, Key_Old;
    // 读取按键值并处理按键状态
    Key_Val = Key_Read();
    Key_Down = Key_Val & (Key_Old ^ Key_Val);    // 检测按键按下
    Key_Up = ~Key_Val & (Key_Old ^ Key_Val);     // 检测按键释放
    Key_Old = Key_Val;
    
    
    // 数字键输入处理 (密码未满8位时接受输入)
    if (input_passwd_index < 8)
    {
        // 检测数字键按下 (100-109对应0-9)
        if (Key_Down >= 100 && Key_Down <= 109)
        {
            input_passwd[input_passwd_index] = Key_Down - 100;  // 转换为实际数字0-9
            input_passwd_index++;                               // 索引指向下一位
        }
    }
    
    
    // 验证密码按键处理
    if (Key_Down == 4)
    {
        // 验证输入的密码是否正确
        if (memcmp(input_passwd, current_passwd, 8) == 0)
        {
            // 密码正确处理
            passwd_error = 0;
            // 可在此添加开门等后续动作
        }
 
        else
        {
            // 密码错误处理
            passwd_error = 1;
            // 可在此添加错误提示等功能
        }
        // 无论正确与否，都清空当前输入
        input_passwd_index = 0;     
        memset(input_passwd, 10, 8); 
    }
    
    
    
    // 清空按键处理
    if (Key_Down == 12)
    {
        input_passwd_index = 0;     // 重置索引
        memset(input_passwd, 10, 8); // 清空输入缓存
    }
}
```

void Seg_Proc()
{
    unsigned char i;
    
    
    // 当有数字输入时进行显示处理
    if (input_passwd_index > 0)
    {
        // 从右向左显示输入的数字(右对齐)
        for (i = 0; i < input_passwd_index; i++)
        {
            // 数组索引映射: 从最右边(7)开始，显示最近输入的数字
            Seg_Buf[7 - i] = input_passwd[input_passwd_index - i - 1];
        }
        
        // 未输入的高位数码管熄灭
        for (; i < 8; i++)
        {
            Seg_Buf[7 - i] = 10;  // 10代表段码全灭
        }
    }
    // 无数据输入时全部熄灭
    else
    {
        for (i = 0; i < 8; i++)
        {
            Seg_Buf[i] = 10;  // 所有数码管熄灭
        }
    }
    
}

## 关于PWM

> 一般是使用定时器0跑任务，定时器1跑pwm在motor的输出，定时器2跑串口 为什么要用定时器1跑pwm而不用定时器0呢，因为通常来说，我们pwm输出 的频率都比较高，那么就意味着pwm的中断的时间肯定是比1ms更短的，所以 我们如果用定时器0去跑中断的话，由于定时器0的优先级比定时器1更高，就 会导致定时器1的中断一直进不去，那么就会导致我们任务直接死掉

- **底层中的led_off()就是和pwm配套的**
