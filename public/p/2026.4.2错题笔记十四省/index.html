/* AD_DA */
void AD_DA()
{
    unsigned char temp_ad;
    temp_ad = Ad_Read(0x01);  // 读取PCF8591通道1的AD值（光敏电阻）
    // 假设用80作为亮暗分割线
    if (Triggered_Acquisition == 0)
    {
        if ((temp_ad < 80) && (Voltage_Light_Value_Old > 80))
        {
            Triggered_Acquisition = 1;       // 触发采集
            Temperature_Acquisitioned = 0;    // 等待采集温度
            Humidity_Acquisitioned = 0;       // 等待采集湿度
            Entry_Data = 0;                   // 等待数据录入
        }
    }
    Voltage_Light_Value_Old = temp_ad;  // 保存当前值为旧值
}

工作原理：

  1. AD值读取：通过 Ad_Read(0x01) 读取 PCF8591 芯片通道1的AD转换值，该通道连接光敏电阻
  2. 是通道1就行，0x01和0x41都可以
  3. 状态判断：
    • 以 80 作为亮暗状态的分割阈值
    • 当当前AD值 < 80（暗）且上一次AD值 > 80（亮）时，检测到从亮到暗的状态变化
  4. 触发机制：
    • 仅在状态变化的瞬间触发采集操作
    • 避免持续触发，提高系统效率
  5. 状态保存：将当前AD值保存到 Voltage_Light_Value_Old 变量，用于下次比较

技术要点：

  • 边缘检测：使用前后两次AD值的比较，实现状态变化的边缘检测
  • 阈值设定：通过经验值 80 作为亮暗状态的判断标准
  • 标志位管理：通过多个标志位控制后续的数据采集流程

这种实现方式可以有效地检测环境光强度的变化，当环境从亮变暗时触发数据采集，适用于需要根据光线变化启动特定操作的场景。

温度传感器初始化问题

DS18B20 初始化等待的原因

while(rd_temperature() == 85); 这行代码的作用是等待 DS18B20 温度传感器初始化完成，主要原因如下：

  1. DS18B20 的特性
  • 默认初始值 ：DS18B20 刚上电时会返回 85°C 的默认值
  • 初始化过程 ：传感器需要时间进行内部校准和准备工作
  • 稳定时间 ：从上电到完全就绪需要一定时间 2. 技术原理
  • 硬件特性 ：DS18B20 在电源接通后，内部电路需要时间稳定
  • 温度转换 ：首次温度转换需要更长时间（最高精度下可达 750ms）
  • 数据可靠性 ：未初始化完成时返回的 85°C 是无效数据 3. 代码实现分析
  • rd_temperature() 函数会执行完整的温度读取流程：
    1. 初始化总线
    2. 发送温度转换命令
    3. 等待转换完成
    4. 读取温度数据
    5. 计算实际温度
  • 当传感器未完全就绪时，会持续返回 85°C
  • 循环会一直执行直到返回值不等于 85°C，确保获取到真实的温度值 4. 重要性
  • 系统稳定性 ：确保系统启动后使用的第一个温度数据是准确的
  • 避免错误数据 ：防止使用 85°C 这个默认值进行后续计算
  • 硬件兼容性 ：适应不同批次、不同状态的 DS18B20 传感器 这种等待机制是 DS18B20 温度传感器使用中的标准做法，确保了温度数据的可靠性和系统的稳定性。

void main()
{
	System_Init();	// 系统底层初始化
	Set_Rtc(ucRtc); // 初始化DS1302时钟时间
	while(rd_temperature() == 85);  //等待DS18B20初始化完成!
	// 各模块初始化
	Timer0_Init();	  // 定时器0（频率计数）初始化
	Scheduler_Init(); // 任务调度器初始化
	Timer1_Init();	  // 定时器1（系统心跳）初始化

	// 主循环
	while (1)
	{
		Scheduler_Run(); // 循环执行任务调度器
	}
}
