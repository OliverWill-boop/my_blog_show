idata unsigned long int uwtick = 0;

typedef struct
{
	void (*task_func)(void);
	unsigned long int rate_ms; // 任务执行周期（毫秒）
	unsigned long int last_ms; // 上次执行时间
} task_t;

idata task_t scheduler_task[] = {
	{Led_Proc, 1, 0},
	{Key_Proc, 10, 0},
	{Seg_Proc, 20, 0},
	{Get_Temperature, 300, 0},
	{Get_Distance, 150, 0},
	{DA_Proc, 100, 0}
};

idata unsigned char task_num;

void Scheduler_Init()
{
	task_num = sizeof(scheduler_task) / sizeof(task_t);
}

void Scheduler_Run()
{
	unsigned char i;
	for (i = 0; i < task_num; i++)
	{
		unsigned long int now_time = uwtick;
		if (now_time >= scheduler_task[i].rate_ms + scheduler_task[i].last_ms)
		{
			scheduler_task[i].last_ms = now_time;
			scheduler_task[i].task_func();
		}
	}
}

void Timer1_Isr(void) interrupt 3    //这里注意一下，数码管和调度器用的都是定时器1
{
	uwtick++;

	seg_pos = (++seg_pos) % 8;  //0~7循环
	if (seg_buf[seg_pos] > 20)
		Seg_Disp(seg_pos, seg_buf[seg_pos] - ',', 1);
	else
		Seg_Disp(seg_pos, seg_buf[seg_pos], 0);
}

void Timer1_Init(void) // 1毫秒@12.000MHz
{
	AUXR &= 0xBF; // 定时器时钟12T模式
	TMOD &= 0x0F; // 设置定时器模式
	TL1 = 0x18;	  // 设置定时初始值
	TH1 = 0xFC;	  // 设置定时初始值
	TF1 = 0;	  // 清除TF1标志
	TR1 = 1;	  // 定时器1开始计时
	ET1 = 1;	  // 使能定时器1中断
	EA = 1;
}

void main()
{
	Scheduler_Init();
	Timer1_Init();
	while (1)
	{
		Scheduler_Run();
	}
}

sbit US_TX=P1^0;  //原理图上会写
sbit US_RX=P1^1;

void Delay12us(void)	//@12.000MHz
{
	unsigned char data i;

	_nop_();
	_nop_();  
	i = 38;  //手动改成38比较好，人家测试出来的
	while (--i);
}

void Ut_Wave_Init()   //发波程序
{
	unsigned char i;
	for(i=0;i<8;i++)
	{
		US_TX=1;
		Delay12us();
		US_TX=0;
		Delay12us();
	}
}

unsigned int Ut_Wave_Data(float speed)
{
	unsigned int time;
	float temp=0;

	CMOD = 0x00;  //死记
	CH=CL=0;   //死记

	Ut_Wave_Init();  //发波

	CR=1;  //开始计时

	while((US_RX==1)&&(CF==0));  //等待返回，跳出这个循环的条件，要么是收到东西了，要么是超出范围定时器溢出了

	CR=0;  //停止计时

	if(CF==0)   //如果没有溢出，也就是收到东西了
	{
		time=CH<<8|CL;
	
		//
	
		temp=(float)time*speed/20000.0f;
		return temp;
	
		//  其实就相当于：return time*0.017;       这里用speed，是因为后面题目有一个温度和速度的公式
	}
	else    //如果没有收到东西
	{
		CF=0;
		return 999;  //官方例程就是这样给的
	}
}

void Get_Distance()
{
	unsigned int temp;
	if (signal_flag)
	{
		// 单次测量模式
		temp = Ut_Wave_Data(speed);
		distance = temp;
		signal_flag = 0;
	}
	else if (summation_flag)
	{
		// 累加测量模式
		temp = Ut_Wave_Data(speed);
		distance_sum += temp;
		summation_index++;

		// 累加测量次数到达结束累加测量
		if (summation_index >= para_num_arr[para_num_index])
		{
			summation_flag = 0;
			summation_index = 0;
		}
	}
}

{
	float temp = 0;
	switch (range_show_mode)
	{
	case 0:
		/* 关闭/待机 */
		temp = 0;
		break;
	case 1:
		/* 单次测量 */

		{
			if (distance <= 20)
				temp = 1;
			else if (distance >= 100)
				temp = 5;
			else
				temp = (float)distance / 20.0f;
		}
		break;
	case 2:
		/* 累加测量 */
		if (summation_flag == 0)
		{
			if (distance_sum <= 20)
				temp = 1;
			else if (distance_sum >= 100)
				temp = 5;
			else
				temp = (float)distance_sum / 20.0f;
		}
		break;
	}
	Da_Write(temp * 51);
}

总结

模块 难度 说明
超声波测距 ★★★ PCA 计时 + 温度补偿声速
累加测量状态机 ★★★★★ 本题最大难点，状态转换多，边界条件复杂
DAC 线性映射 ★★ 公式推导 + 分段处理
参数往复切换 ★★ 理解往复 vs 循环的区别
界面显示管理 ★★★ 三个界面 + 多种显示状态
LED 状态控制 ★★★ L2 累加中不变 + L3 定时闪烁
按键屏蔽逻辑 ★★★★ 不同界面/状态下的互斥规则多
整体调度架构 ★★★ 多任务协调，阻塞操作的影响
  1. 先搭框架：任务调度器 + 界面切换 + 数码管显示（基础分）
  2. 再做温度和单次测距：DS18B20 + 超声波单次（中等分）
  3. 然后做 DAC 和 LED：输出控制（稳定分）
  4. 最后攻克累加测量：状态机 + 按键屏蔽（高分区）
